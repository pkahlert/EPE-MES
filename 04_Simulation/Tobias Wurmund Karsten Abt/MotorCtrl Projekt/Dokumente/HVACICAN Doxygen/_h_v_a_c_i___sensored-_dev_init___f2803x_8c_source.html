49 // sources to minimize power consumption

59 // SYSTEM CLOCK speed based on internal oscillator = 10 MHz
60 // 0xC = 60 MHz (12)
61 // 0xB = 55 MHz (11)
62 // 0xA = 50 MHz (10)
63 // 0x9 = 45 MHz (9)
64 // 0x8 = 40 MHz (8)
65 // 0x7 = 35 MHz (7)
66 // 0x6 = 30 MHz (6)
67 // 0x5 = 25 MHz (5)
68 // 0x4 = 20 MHz (4)
69 // 0x3 = 15 MHz (3)
70 // 0x2 = 10 MHz (2)
71 
72  PLLset(0xC); // choose from options above

77  PieVectTableInit();

89 // from TI reserved OTP into the appropriate trim registers, occurs automatically
90 // in the Boot ROM. If the boot ROM code is bypassed during the debug process, the
91 // following function MUST be called for the ADC and oscillators to function according
92 // to specification.

94  SysCtrlRegs.PCLKCR0.bit.ADCENCLK = 1; // Enable ADC peripheral clock
95  (*Device_cal)(); // Auto-calibrate from TI OTP

443 // This function initializes the PLLCR register.
444 //void InitPll(Uint16 val, Uint16 clkindiv)
445 void PLLset(Uint16 val)
446 {
447  volatile Uint16 iVol;
448 
449  // Make sure the PLL is not running in limp mode
450  if (SysCtrlRegs.PLLSTS.bit.MCLKSTS != 0)
451  {
452  EALLOW;
453  // OSCCLKSRC1 failure detected. PLL running in limp mode.
454  // Re-enable missing clock logic.
455  SysCtrlRegs.PLLSTS.bit.MCLKCLR = 1;
456  EDIS;
457  // Replace this line with a call to an appropriate
458  // SystemShutdown(); function.
459  asm(" ESTOP0"); // Uncomment for debugging purposes
460  }
461 
462  // DIVSEL MUST be 0 before PLLCR can be changed from
463  // 0x0000. It is set to 0 by an external reset XRSn
464  // This puts us in 1/4
465  if (SysCtrlRegs.PLLSTS.bit.DIVSEL != 0)
466  {
467  EALLOW;
468  SysCtrlRegs.PLLSTS.bit.DIVSEL = 0;
469  EDIS;
470  }
471 
472  // Change the PLLCR
473  if (SysCtrlRegs.PLLCR.bit.DIV != val)
474  {
475 
476  EALLOW;
477  // Before setting PLLCR turn off missing clock detect logic
478  SysCtrlRegs.PLLSTS.bit.MCLKOFF = 1;
479  SysCtrlRegs.PLLCR.bit.DIV = val;
480  EDIS;
481 
482  // Optional: Wait for PLL to lock.
483  // During this time the CPU will switch to OSCCLK/2 until
484  // the PLL is stable. Once the PLL is stable the CPU will
485  // switch to the new PLL value.
486  //
487  // This time-to-lock is monitored by a PLL lock counter.
488  //
489  // Code is not required to sit and wait for the PLL to lock.
490  // However, if the code does anything that is timing critical,
491  // and requires the correct clock be locked, then it is best to
492  // wait until this switching has completed.
493 
494  // Wait for the PLL lock bit to be set.
495  // The watchdog should be disabled before this loop, or fed within
496  // the loop via ServiceDog().
497 
498  // Uncomment to disable the watchdog
499  WDogDisable();
500 
501  while(SysCtrlRegs.PLLSTS.bit.PLLLOCKS != 1) {}
502 
503  EALLOW;
504  SysCtrlRegs.PLLSTS.bit.MCLKOFF = 0;
505  EDIS;
506  }
507 
508  //divide down SysClk by 2 to increase stability
509  EALLOW;
510  SysCtrlRegs.PLLSTS.bit.DIVSEL = 2;
511  EDIS;
512 }
