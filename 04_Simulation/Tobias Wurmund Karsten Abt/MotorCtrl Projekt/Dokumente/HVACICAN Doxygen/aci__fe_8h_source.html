100 /* The current model section (Classical Rotor Flux Vector Control Equation)*/ \
101  v.FluxDrE = _IQmpy(v.K1,v.FluxDrE) + _IQmpy(v.K2,v.IDsE); \
102  \
103 /* Inverse park transformation on the rotor flux from the current model*/ \
104  v.FluxDrS = _IQmpy(v.FluxDrE,v.Cosine); \
105  v.FluxQrS = _IQmpy(v.FluxDrE,v.Sine); \
106  \
107 /* Compute the stator flux based on the rotor flux from current model*/ \
108  v.FluxDsS = _IQmpy(v.K3,v.FluxDrS) + _IQmpy(v.K4,v.IDsS); \
109  v.FluxQsS = _IQmpy(v.K3,v.FluxQrS) + _IQmpy(v.K4,v.IQsS); \
110  \
111 /* Conventional PI controller section */ \
112  v.Error = v.PsiDsS - v.FluxDsS; \
